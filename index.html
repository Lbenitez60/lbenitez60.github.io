Some very simple but useful shell functions:

#Command help from cheat.sh. Takes the mysterious command as a string.
cheat() {
 if [[ -n $* ]]; then
  curl cheat.sh/$*
 else
  curl cheat.sh
 fi
}

#Add, commit, and push. Requires commit message as a string.
gall(){
 if [[ -n $* ]]; then
  git add *; git commit -m $*; git push
 else
  print "Please use a commit message!"
 fi
}
